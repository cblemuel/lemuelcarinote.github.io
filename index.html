// HTML
This div will be displayed when the form has a value or the customer is typing.
// JavaScript const myInput = document.getElementById("myInput"); const myDiv = document.getElementById("myDiv"); // Listen for changes to the value of the input field myInput.addEventListener("input", () => { // If the value of the input field is not empty, display the div if (myInput.value) { myDiv.style.display = "block"; } }); // Listen for keystrokes in the input field myInput.addEventListener("keydown", () => { // If the user is typing, display the div if (myInput.value.length > 0) { myDiv.style.display = "block"; } });
